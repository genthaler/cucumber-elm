module GherkinRecord exposing (..)


type alias Feature =
    { feature : String, asA : String, inOrderTo : String, iWantTo : String, background : Background, scenarios : List Scenario }


type alias Background =
    { background : String, steps : List Step }


type Scenario
    = Scenario { scenario : String, steps : List Step }
    | ScenarioOutline { scenario : String, steps : List Step, examples : DataTable }


type Step
    = Given String StepArg
    | When String StepArg
    | Then String StepArg
    | And String StepArg


type StepArg
    = NoArg
    | DataTable DataTable
    | DocString String


type alias DataTable =
    List (List String)
